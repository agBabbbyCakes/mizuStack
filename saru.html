deploy.py
Deploy a Contract
from ape import project, accounts

# Load your account
account = accounts.load('dev')

# Deploy your contract
contract = project.MyContract.deploy(sender=account)

print(f"Contract deployed at: {contract.address}")
Interact with a Contract
from ape import Contract

# Load contract by address
contract = Contract("0x123...")

# Call a view function
balance = contract.balanceOf(account)

# Send a transaction
receipt = contract.transfer(
    recipient,
    amount,
    sender=account
)
Batch Multiple Calls
from ape_ethereum import multicall

# Create a multicall
call = multicall.Call()

# Add multiple calls
call.add(contract1.balanceOf, account)
call.add(contract2.balanceOf, account)

# Execute all calls at once
results = call()
Publish Contract Source
from ape import project

# Deploy with source verification
contract = project.MyContract.deploy(
    sender=account,
    publish=True  # Automatically verify source
)
Write Tests
import pytest
from ape import accounts

@pytest.fixture
def owner(accounts):
    return accounts[0]

def test_deploy(owner, project):
    contract = project.MyContract.deploy(sender=owner)
    assert contract.address is not None
